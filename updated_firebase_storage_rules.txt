rules_version = '2';

service firebase.storage {
  match /b/{bucket}/o {
    
    // User drafts (private to owner)
    match /user_drafts/{userId}/{allPaths=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Public templates
    match /public_templates/{allPaths=**} {
      allow read: if true; // everyone can read
      allow write: if request.auth != null && (request.auth.uid=="LTRhJmbufLQP0hrJj5xNKwOfDc53" || request.auth.uid=="aP3FVBY7kWgBnJorqVrYha3lFaa2"); // only you
    }

    // Fonts (public read access for all users)
    match /fonts/{allPaths=**} {
      allow read: if true; // everyone can read/download fonts
      allow write: if request.auth != null && (request.auth.uid=="LTRhJmbufLQP0hrJj5xNKwOfDc53" || request.auth.uid=="aP3FVBY7kWgBnJorqVrYha3lFaa2"); // only you can upload/manage fonts
    }
  }
}
